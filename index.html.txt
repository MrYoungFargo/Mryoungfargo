  // Load cart from localStorage

  function loadCart() {
    const saved = localStorage.getItem("duets_cart");
    if (saved) {
      cart = JSON.parse(saved);
    } else {
      cart = [];
    }
    updateCartUI();
  }
  function saveCart() {
    localStorage.setItem("duets_cart", JSON.stringify(cart));
    updateCartUI();
  }


  function updateCartUI() {
    const cartCountSpan = document.getElementById("cartCount");
    const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
    cartCountSpan.innerText = totalItems;


    const cartContainer = document.getElementById("cartItemsList");
    const totalSpan = document.getElementById("cartTotalPrice");
    let totalPrice = 0;
    if (!cartContainer) return;
    if (cart.length === 0) {
      cartContainer.innerHTML = '<div class="empty-cart-msg">Your cart is empty.<br><i class="fas fa-shopping-bag"></i></div>';
      totalSpan.innerText = "$0.00";
      return;
    }
    let html = "";
    cart.forEach((item, idx) => {
      const itemTotal = item.price * item.quantity;
      totalPrice += itemTotal;
      html += `
        <div class="cart-item">
          <div><strong>${item.name}</strong><br>$${item.price} x ${item.quantity}</div>
          <div><button class="remove-item" data-id="${item.id}" style="background:var(--red); border:none; color:white; padding:6px 12px; border-radius:40px; cursor:pointer;">Remove</button></div>
        </div>
      `;
    });
    cartContainer.innerHTML = html;
    totalSpan.innerText = `$${totalPrice.toFixed(2)}`;
    document.querySelectorAll(".remove-item").forEach(btn => {
      btn.addEventListener("click", (e) => {
        const id = parseInt(btn.getAttribute("data-id"));
        removeFromCart(id);
      });
    });
  }


  function addToCart(product) {
    const existing = cart.find(item => item.id === product.id);
    if (existing) {
      existing.quantity += 1;
    } else {
      cart.push({ ...product, quantity: 1 });
    }
    saveCart();
    showCartNotification();
  }


  function removeFromCart(productId) {
    cart = cart.filter(item => item.id !== productId);
    saveCart();
  }

  }
